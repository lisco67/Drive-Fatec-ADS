class Circunferencia
{

     private float raio;
     
     public Circunferencia()
     {
     }
     
     public Circunferencia(float r)
     {  
           raio = r;
     }       


     public diametro()
     {
        return 2*raio;
     }


     public area()
     {
         return 3.14*raio*raio;
     }
}
